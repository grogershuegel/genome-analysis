  group_by(Country_Region) %>%
  summarise(Confirmed = sum(Confirmed)) %>% 
  arrange(desc(Confirmed)) %>% 
  slice(1:10) %>% 
  ggplot(aes(x=Confirmed, y=reorder(Country_Region, Confirmed)))+
  geom_point()+
  ggtitle("10 Countries with Most COVID-19 Cases on March 11")+

Time Series Graphs

time_series_long_joined_counts %>% 
  group_by(Country_Region, Report_Type, Date) %>% 
  summarise(Counts = sum(Counts)) %>% 
  filter(Country_Region == "Italy") %>% 
  filter(Report_Type %in% c("Confirmed", "Deaths", "Recovered")) %>% 
  ggplot(aes(x=Date, y=Counts, fill= Report_Type, color=Report_Type))+
  geom_point()+
  geom_line()+
  ggtitle("COVID-19 Cases in Italy")

time_series_long_joined %>% 
  group_by(Country_Region,Date) %>% 
  summarise_at(c("Confirmed", "Deaths", "Recovered"), sum) %>% 
  filter (Country_Region %in% c("Spain","France","Italy", 
                                "Germany")) %>% 
  ggplot(aes(x = Date,  y = log2(Deaths), color = Country_Region)) + 
  geom_point() +
  geom_line() +
  labs(color= "Country")+
  ggtitle("COVID-19 Deaths in European Countries")
